#### Usage
```ts
import {ReactFlow, Background, Controls, Node, Edge} from '@xyflow/react';  
import '@xyflow/react/dist/style.css';
```

```ts
function Page() {  
    const nodes:Node[] = [  
        {  
            id:"1",  
            data : {  
                label : "Node Label 1"  
            },  
            position : {x:0,y:0}  
        },  
        {  
            id:"2",  
            data : {  
                label : "Node Label 2"  
            },  
            position : {x:0,y:100}  
        }  
    ];  
  
    const edges:Edge[] = [  
        {  
            id:"1-1",  
            source:"1",  
            target:"2",  
            animated:true // Creates animated flow lines
        }  
    ];  
    return (  
        <div className={"h-screen w-full bg-gradient-to-tr from-zinc-950 via-zinc-900 to-zinc-950"}>  
            <ReactFlow nodes={nodes} edges={edges}>  
                <Background/>  
                <Controls/>  
                <MiniMap/>
            </ReactFlow>  
        </div>  
    );  
}  
  
export default Page;
```

#### Adding Interactions
```ts
"use client";  
  
import React from "react";  
import {  
    ReactFlow,  
    Background,  
    Controls,  
    Node,  
    Edge, useNodesState, useEdgesState,  
    Handle,Position  
} from "@xyflow/react";  
import "@xyflow/react/dist/style.css";  
  
function CustomNode({ data }: { data: { label: string } }) {  
    return (  
        <div className="relative px-4 py-2 bg-zinc-800/50 backdrop-blur-2xl text-white rounded-xl shadow-md border border-zinc-600">  
            {data.label}  
  
            {/* Handles */}  
            <Handle  
                type="target"  
                position={Position.Top}  
                className="w-3 h-3 !bg-cyan-400 border-none"  
            />  
            <Handle  
                type="source"  
                position={Position.Bottom}  
                className="w-3 h-3 !bg-cyan-400 border-none"  
            />  
        </div>  
    );  
}  
  
function Page() {  
    const initialNodes: Node[] = [  
        {  
            id: "1",  
            // type: "custom",  
            data: { label: "Node Label 1" },  
            position: { x: 0, y: 0 },  
        },  
        {  
            id: "2",  
            // type: "custom",  
            data: { label: "Node Label 2" },  
            position: { x: 0, y: 100 },  
        },  
    ];  
  
    const initialEdges: Edge[] = [  
        {  
            id: "1-1",  
            source: "1",  
            target: "2",  
            animated: true,  
        },  
    ];  
  
  
    const [nodes, setNodes, onNodesChange] = useNodesState(initialNodes);  
    const [edges, setEdges, onEdgesChange] = useEdgesState(initialEdges);  
  
    return (  
        <div className="h-screen w-full bg-gradient-to-tr from-zinc-950 via-zinc-900 to-zinc-950">  
            <ReactFlow nodes={nodes} edges={edges} onNodesChange={onNodesChange} onEdgesChange={onEdgesChange} nodeTypes={{ custom: CustomNode }}>  
                <Background />  
                <Controls />  
            </ReactFlow>  
        </div>  
    );  
}  
  
export default Page;
```

#### Connecting nodes with edges
```ts
"use client";  
  
import React, {useCallback} from "react";  
import {  
    ReactFlow,  
    Background,  
    Controls,  
    Node,  
    Edge, useNodesState, useEdgesState,  
    Handle, Position, Connection, addEdge  
} from "@xyflow/react";  
import "@xyflow/react/dist/style.css";  
  
function CustomNode({ data }: { data: { label: string } }) {  
    return (  
        <div className="relative px-4 py-2 bg-zinc-800/50 backdrop-blur-2xl text-white rounded-xl shadow-md border border-zinc-600">  
            {data.label}  
  
            {/* Handles */}  
            <Handle  
                type="target"  
                position={Position.Top}  
                className="w-3 h-3 !bg-cyan-400 border-none"  
            />  
            <Handle  
                type="source"  
                position={Position.Bottom}  
                className="w-3 h-3 !bg-cyan-400 border-none"  
            />  
        </div>  
    );  
}  
  
function Page() {  
    const initialNodes: Node[] = [  
        {  
            id: "1",  
            // type: "custom",  
            data: { label: "Node Label 1" },  
            position: { x: 0, y: 0 },  
        },  
        {  
            id: "2",  
            // type: "custom",  
            data: { label: "Node Label 2" },  
            position: { x: 0, y: 100 },  
        },  
    ];  
  
    const initialEdges: Edge[] = [  
        {  
            id: "1-1",  
            source: "1",  
            target: "2",  
            animated: true,  
        },  
    ];  
  
  
    const [nodes, setNodes, onNodesChange] = useNodesState(initialNodes);  
    const [edges, setEdges, onEdgesChange] = useEdgesState(initialEdges);  
  
    const onConnect = useCallback((connection:Connection) => {  
        const edge = {...connection, animated:"true", id:`${edges.length+1}`};  
        setEdges((prevEdges)=>addEdge(edge,prevEdges));  
    },[]);  
  
    return (  
        <div className="h-screen w-full bg-gradient-to-tr from-zinc-950 via-zinc-900 to-zinc-950">  
            <ReactFlow nodes={nodes} edges={edges} onConnect={onConnect} onNodesChange={onNodesChange} onEdgesChange={onEdgesChange} nodeTypes={{ custom: CustomNode }}>  
                <Background />  
                <Controls />  
            </ReactFlow>  
        </div>  
    );  
}  
  
export default Page;
```
